# 短調非減少の答えでの値Kの範囲

```cpp

pair<int,int> bs(int k){

    int big,small;
    int l = 0;
    int r = INF;
    while(l+1 < r){    // find big
        int mid = (l+r)/2;
        int res = test(mid);
        if(res >= k){
            l = mid;
        }
        else{
            r = mid;
        }
    }
    big = l;
    l = 0;
    r = INF;
    while(l+1 < r){    // find small
        int mid = (l+r)/2;
        int res = test(mid);
        if(res > k){
            l = mid;
        }
        else{
            r = mid;
        }
    }    
    small = r;
    return {small,big};
}

```
