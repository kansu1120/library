# 累積和をやるためのライブラリ

やるだけ

```cpp
template<typename T>
class rui{
    public:
    vector<T> v;
    rui(const vector<T>& a){
        v = a;
        for(int i = 1;i < a.size();i++){
            v[i] += v[i-1];
        }
    }
    T query(int l,int r){
        if(l == 0){
            return v[r];
        }
        else{
            return v[r] - v[l-1];
        }
    }
};
```
